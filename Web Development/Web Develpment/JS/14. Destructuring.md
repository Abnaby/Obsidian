- Is a JavaScript expression that allows us to extract data from arrays,
    objects, and maps and set them into new, distinct variables.
### With Arrays
- Try it
```js
let a, b, rest;
[a, b] = [10, 20];

console.log(a);
// Expected output: 10

console.log(b);
// Expected output: 20

[a, b, ...rest] = [10, 20, 30, 40, 50];
console.log(rest);
// Expected output: Array [30, 40, 50]
```
- Syntax For Arrays 
```js
// Instant Decleration and Defention 
let [a, b] = array;
let [a, , b] = array;
let [a = aDefault, b] = array;
let [a, b, ...rest] = array;
let [a, , b, ...rest] = array;
let [a, b, ...{ pop, push }] = array;
let [a, b, ...[c, d]] = array;
// Variables are declered before 
let a, b, a1, b1, c, d, rest, pop, push;
[a, b] = array;
[a, , b] = array;
[a = aDefault, b] = array;
[a, b, ...rest] = array;
[a, , b, ...rest] = array;
[a, b, ...{ pop, push }] = array;
[a, b, ...[c, d]] = array;
```
- Example
```js
let a = 1;
let b = 2;
let c = 3;
let d = 4;

let myFriends = ["Ahmed", "Sayed", "Ali", "Maysa"];

[a = "A", b, c, d, e = "Osama"] = myFriends;

console.log(a); // Ahmed
console.log(b); // Sayed
console.log(c); // Ali
console.log(d); // Maysa
console.log(e); //Osama Due to Default value
/************************** SKIPPING ELEMENTS IN ARRAY *************/
let [, y, , z] = myFriends;
console.log(y);
console.log(z);

```
- Nested Arrays 
```javascript

let myFriends = ["Ahmed", "Sayed", "Ali", ["Shady", "Amr", ["Mohamed", "Gamal"]]];
// هتاخد شيب الاراي وتمسح اللي مش عاوزو 
let [, , , [a, , [, b]]] = myFriends;

console.log(a); // Shady
console.log(b); // Gamal
```
- Swap Example 
```js
/*
  Destructuring
  - Destructuring Array => Swapping Variables
*/

let book = "Video";
let video = "Book";

[book, video] = [video, book];

console.log(book); // Book
console.log(video); // Video
```
### With Objects
- Notes
	- Select by key name 
- Try It
```js
const obj = { a: 1, b: 2 };
const { a, b } = obj;
// is equivalent to:
// const a = obj.a;
// const b = obj.b;
```
- Syntax For Objects
```js
// Instant Decleration and Defention 
const { a, b } = obj;
const { a: a1, b: b1 } = obj;
const { a: a1 = aDefault, b = bDefault } = obj;
const { a, b, ...rest } = obj;
const { a: a1, b: b1, ...rest } = obj;
const { [key]: a } = obj;

// Variables are declered before 
let a, b, a1, b1, c, d, rest, pop, push;
({ a, b } = obj); // parentheses are required
({ a: a1, b: b1 } = obj);
({ a: a1 = aDefault, b = bDefault } = obj);
({ a, b, ...rest } = obj);
({ a: a1, b: b1, ...rest } = obj);
```
- Simple Example to skip key name
```js
const user = {
  theName: "Osama",
  theAge: 39,
  theTitle: "Developer",
  theCountry: "Egypt",
};


// ({ theName, theAge, theTitle, theCountry } = user);
const { theName, theAge, theCountry } = user;

console.log(theName);    // Osama
console.log(theAge);     // 39
console.log(theCountry); // Egypt
```
- Variables Declared Previous
```js
const numbers = [];
const obj = { a: 1, b: 2 };
({ a: numbers[0], b: numbers[1] } = obj); // // parentheses are required
// The properties `a` and `b` are assigned to properties of `numbers`
// numbers[0] = 1
// numbers[1] = 2
```
- Simple Example with changed variable key 
```js
const user = {
  theName: "Osama",
  theAge: 39,
  theTitle: "Developer",
  theCountry: "Egypt",
};


// ({ theName, theAge, theTitle, theCountry } = user);
const { theName:a, theAge:b, theCountry:c, NewProb:d = "Default" } = user;

console.log(a); // Osama
console.log(b); // 39
console.log(c); // Egypt
console.log(d); // Default
```
- Nested Objects 
```js
/*
  Destructuring
  - Destructuring Object
  --- Naming The Variables
  --- Add New Property
  --- Nested Object
  --- Destructuring The Nested Object Only
*/

const user = {
  theName: "Osama",
  theAge: 39,
  theTitle: "Developer",
  theCountry: "Egypt",
  theColor: "Black",
  skills: {
    html: 70,
    css: 80,
  },
};

// Destructuring 
const {
  theName: n,
  theAge: a,
  theCountry, // keyName
  theColor: co = "Red",
  skills: { html: h, css },
} = user;

console.log(n);   // Osama
console.log(a);   // 39
console.log(theCountry); // Egypt
console.log(co); // Red
console.log(`My HTML Skill Progress Is ${h}`); // 70
console.log(`My CSS Skill Progress Is ${css}`); // 80

const { html: skillOne, css: skillTwo } = user.skills;

console.log(`My HTML Skill Progress Is ${skillOne}`); //70
console.log(`My CSS Skill Progress Is ${skillTwo}`);  //80
```
### Destructuring Function Parameters
```js
/*
  Destructuring
  - Destructuring Function Parameters
*/

const user = {
  theName: "Osama",
  theAge: 39,
  skills: {
    html: 70,
    css: 80,
  },
};

showDetails(user);

function showDetails({ theName: n, theAge: a, skills: { css: c } }) {
  console.log(`Your Name Is ${n}`);
  console.log(`Your Age Is ${a}`);
  console.log(`Your CSS Skill Progress Is ${c}`);
  /*
	Your Name Is Osama
	Your Age Is 39
	Your CSS Skill Progress Is 80
  */
}
```
