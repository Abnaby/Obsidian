### Named Export  
1. Define Modules in HTML file
```html 
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <link rel="stylesheet" href="main.css" />
    <title>Learn JavaScript</title>
  </head>
  <body>
    <script src="main.js" type="module"></script> 
    <script src="app.js" type="module"></script>
  </body>
</html>
```
2. Export What u want 
```js
// Main.js
let a = 10;
let arr = [1, 2, 3, 4];

function saySomething() {
  return `Something`;
}
export { a, arr, saySomething };
```
3. Import what u need 
```js
// app.js 
import { a, arr, saySomething as s } from "./main.js";

console.log(a);     // 10
console.log(arr);   // [1, 2, 3 ,4]
console.log(s());   // "Something"
```
### Default Export 
```js
// Main.js
let a = 10;
let arr = [1, 2, 3, 4];

function saySomething() {
  return `Something`;
}

export { a as myNumber, arr, saySomething };

// Default could be normal/anonymous fcn 
export default function () {
  return `Hello`;
}
```

```js
// app.js 
import anyNameForDefaultExport , { a, arr, saySomething as s } from "./main.js";

console.log(a);     // 10
console.log(arr);   // [1, 2, 3 ,4]
console.log(s());   // "Something"
console.log(anyNameForDefaultExport);   // "Hello"
```
### Import all 
```js
// Main.js
let a = 10;
let arr = [1, 2, 3, 4];

function saySomething() {
  return `Something`;
}

export { a as myNumber, arr, saySomething };

// Default could be normal/anonymous fcn 
export default function () {
  return `Hello`;
}
```

```js
// app.js 
import * as all from "./main.js";
import defaultFunction from "./main.js"; // Import the default function

console.log(all.a);     // 10
console.log(all.arr);   // [1, 2, 3 ,4]
console.log(all.s());   // "Something"
console.log(defaultFunction());  // "Hello"
```
