## DOM Deal With Children's

```js
/*
  DOM [Deal With Childrens]
  - children
  - childNodes
  - firstChild
  - lastChild
  - firstElementChild
  - lastElementChild
*/

let myElement = document.querySelector("div");

console.log(myElement);
console.log(myElement.children);
console.log(myElement.children[0]);
console.log(myElement.childNodes); // Consider Spaces, \n, Comments, etc ...
console.log(myElement.childNodes[0]);

// myElement.firstChild === myElement.childNodes[0]
console.log(myElement.firstChild); // Consider Spaces, \n, Comments, etc ...
console.log(myElement.lastChild);

console.log(myElement.firstElementChild); //Consider html elements only
console.log(myElement.lastElementChild);
```

## DOM Deal With Sibling

```js
/*
  DOM [Traversing]
  - nextSibling
  - previousSibling
  - nextElementSibling
  - previousElementSibling
  - parentElement
*/
let myPragraph = document.querySelector(".myPragraph");

console.log("nextSibling is ");
console.log(myPragraph.nextSibling) // Consider Spaces, \n, Comments, etc ...

console.log("previousSibling is ");
console.log(myPragraph.previousSibling) // Consider Spaces, \n, Comments, etc ...

console.log("nextElementSibling is ");
console.log(myPragraph.nextElementSibling) //Consider html elements only

console.log("previousElementSibling is ");
console.log(myPragraph.previousElementSibling) //Consider html elements only

console.log("parentElement is ");
console.log(myPragraph.parentElement)
```
