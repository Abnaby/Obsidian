### Notes 
- Run its code when required 
- Return Special Object (Generator Object)
- Generators Are Iterable
### Syntax 
```js
function* name(param0) {
  statements
}
function* name(param0, param1) {
  statements
}
function* name(param0, param1, /* â€¦, */ paramN) {
  statements
}

```

### Examples 
#### Simple Usage 
- Each time call "next().value" it will gives next yield
```javascript
function* generator(i) {
  yield i;
  yield i + 10;
}

const gen = generator(10);

console.log(gen.next().value);
// Expected output: 10

console.log(gen.next().value);
// Expected output: 20

console.log(typeof gen);
// "object"
```
#### Iterate on it 
```js
function* generateNumbers() {
  yield 1;
  console.log("Hello After Yield 1");
  yield 2;
  yield 3;
  yield 4;
}

let generator = generateNumbers();

console.log(typeof generator); // "object"
console.log(generator);        // [object Generator]

console.log(generator.next()); // Object { value: 1, done: false }
console.log(generator.next()); // "Hello After Yield 1"
console.log(generator.next()); // Object { value: 2, done: false }

for (let value of generateNumbers()) {
  console.log(value); // 1 "Hello After Yield 1" , 2 , 3 ,4 
}

for (let value of generator) {
  console.log(value); // 1 "Hello After Yield 1" , 2 , 3 ,4 
}

```
#### Example on Delegate Generator with stop
```javascript
function* generateNums() {
  yield 1;
  yield 2;
  yield 3;
}

function* generateLetters() {
  yield "A";
  yield "B";
  yield "C";
}

function* generateAll() {
  yield* generateNums();
  yield* generateLetters();
  yield* [4, 5, 6];
}

let generator = generateAll();

console.log(generator.next());      // > Object { value: 1, done: false }
console.log(generator.next());      // > Object { value: 2, done: false }
console.log(generator.next());      // > Object { value: 3, done: false }
console.log(generator.next());      // > Object { value: "A", done: false }
console.log(generator.next());      // > Object { value: "B", done: false }
console.log(generator.next());      // > Object { value: "C", done: false }
console.log(generator.return("Z")); // > Object { value: "Z", done: true }
console.log(generator.next());      // > Object { value: undefined, done: true }
console.log(generator.next());      // > Object { value: undefined, done: true }
console.log(generator.next());      // > Object { value: undefined, done: true }
```
#### Example on Delegate Generator without stop
```javascript
function* generateNums() {
  yield 1;
  yield 2;
  yield 3;
}

function* generateLetters() {
  yield "A";
  yield "B";
  yield "C";

function* generateAll() {
  yield* generateNums();
  yield* generateLetters();
  yield* [4, 5, 6];

let generator = generateAll();

console.log(generator.next());      // > Object { value: 1, done: false }

```
#### Generate Infinite Numbers
```javascript
function* generateNumbers() {
  let index = 0;

  while (true) {
    yield index++;
  }
}

let generator = generateNumbers();

console.log(generator.next()); // > Object { value: 0, done: false }
console.log(generator.next()); // > Object { value: 1, done: false }
console.log(generator.next()); // > Object { value: 2, done: false }
console.log(generator.next()); // > Object { value: 3, done: false }
```
#### Generator as an object method
```js
class Foo {
  *generator() {
    yield 1;
    yield 2;
    yield 3;
  }
}

const f = new Foo();
const gen = f.generator();

console.log(gen.next()); // { value: 1, done: false }
console.log(gen.next()); // { value: 2, done: false }
console.log(gen.next()); // { value: 3, done: false }
console.log(gen.next()); // { value: undefined, done: true }
```
#### Generator as an object property
```js
const someObj = {
  *generator() {
    yield "a";
    yield "b";
  },
};

const gen = someObj.generator();

console.log(gen.next()); // { value: 'a', done: false }
console.log(gen.next()); // { value: 'b', done: false }
console.log(gen.next()); // { value: undefined, done: true }
```
